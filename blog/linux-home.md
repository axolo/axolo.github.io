|   |-- crontab                             #   自动任务
|   `-- certs                               #   证书
|-- projects                                # 项目
|   |-- github                              #   GitHUB
|   `-- coding                              #   Coding.net

|   |-- node-v9.11.2-linux-x86.tar.gz       #   Node.js
